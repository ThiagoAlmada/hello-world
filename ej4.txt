 public static boolean evaluaNumero(int numero){
       
        
        boolean condicion=false;
        int resto;
        
        int contador=0;
        
                
        for (int i = 1; i <= numero; i++) {
            resto=numero%i;
            if (resto==0) {
                contador++;
            } 
        }
        if (contador==2) {
            condicion=true;
        }
        return condicion;
        
    }
